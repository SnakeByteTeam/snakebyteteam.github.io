\subsubsection{MyVimar Integration}

% ─── dto in ───
\addAtr{+}{redirect\_url: string}{}
\makeAbCl{PlantAuthDto}{DTO di input che trasporta le credenziali di autenticazione per un impianto MyVimar}

\addAtr{+}{ticket: string}{}
\makeAbCl{PrepareOAuthTicketDto}{DTO di input per la preparazione del ticket OAuth, contenente i dati necessari all'avvio del flusso di autorizzazione}

% ─── dto out ───
\addAtr{+}{isLinked: boolean}{Indica se l'account MyVimar è collegato all'utente corrente}
\addAtr{+}{email: string}{Email dell'account MyVimar collegato}
\addAtr{+}{success: boolean}{Indica se l'operazione di recupero stato è avvenuta con successo}
\makeCl{MyVimarAccountStatusDto}{DTO di output che trasporta lo stato del collegamento dell'account MyVimar dell'utente}

\addAtr{+}{accessToken: string}{Token di accesso restituito dall'API MyVimar}
\addAtr{+}{refreshToken: string}{Token di refresh per il rinnovo del token di accesso}
\addAtr{+}{expiresIn: number}{Durata in secondi della validità del token di accesso}
\addAtr{+}{email: string}{Email dell'account MyVimar associato ai token}
\makeCl{TokensDto}{DTO di output che trasporta la coppia di token restituita dall'API MyVimar dopo il completamento del flusso OAuth}

% ─── model ───
\addAtr{-}{accessToken: string}{Token di accesso corrente}
\addAtr{-}{refreshToken: string}{Token di refresh per il rinnovo del token di accesso}
\addAtr{-}{expiresAt: Date}{Timestamp di scadenza del token di accesso}
\addMet{+}{getAccessToken() : string}{Restituisce il token di accesso}
\addMet{+}{getRefreshToken() : string}{Restituisce il token di refresh}
\addMet{+}{getExpiresAt() : Date}{Restituisce il timestamp di scadenza del token}
\makeCl{TokenPair}{Oggetto di dominio che rappresenta una coppia di token OAuth (accesso e refresh) con la relativa scadenza}
