\begin{frame}{Finora}
  In ordine di tempo:
  \begin{enumerate}
    \item Prima stesura Norme di Progetto;
    \item Allineamento conoscenze di base;
    \item Prime analisi capitolati;
    \item Domande alle Proponenti.
  \end{enumerate}
\end{frame}

\begin{frame}{Cosa faremo}
  In ordine di tempo:
  \begin{enumerate}
    \item Terminare l'analisi dei capitolati;
    \item Fissare la scelta del capitolato;
    \item Stesura Lettera di Candidatura;
    \item Stesura Dichiarazione Impegni.
  \end{enumerate}
\end{frame}

\begin{frame}{Difficoltà e dubbi incontrati}

    \item Che convenzione adottare per versionare i documenti;
    \item Come tracciare le attività a venire e le decisioni prese durante 
    le riunioni del gruppo.
  \end{itemize}
\end{frame}
